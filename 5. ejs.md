### installing ejs, transferring data from node to .ejs file

##### index.js
```js
const express = require("express");

const app = express();

app.set('view engine', 'ejs'); // to make node acknowledge our template engine.

app.get('', (req, res) => {
  const user = {
    name: "mayank", 
    phoneNumber: "100", 
    city: "Delhi"
  }
  res.render('profile', {user}); // not the send or sendFile
});

app.listen(3000);

```


##### profile.html

```html
<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
</head>

<body>
  <h1>Welcome <%= user.name %>
  </h1>
  <h1>phoneNumber <%= user.phoneNumber %>
  </h1>
  <h1>City <%= user.city%>
  </h1>

</body>

</html>

```


### looping in ejs

##### index.js

```js
const express = require("express");

const app = express();

app.set('view engine', 'ejs');

app.get('', (req, res) => {
  const user = {
    name: "mayank",
    phoneNumber: "100",
    city: "Delhi",
    skills: ['a', 'b', 'c', 'd'],
  }
  res.render('profile', { user });
});

app.listen(4000);
```

##### profile.ejs

```js
<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
</head>

<body>
  <h1>Welcome <%= user.name %>
  </h1>
  <h1>phoneNumber <%= user.phoneNumber %>
  </h1>
  <h1>City <%= user.city%>
  </h1>

  <ul>Skills:
    <% user.skills.forEach((value, index)=> { %>
      <li>
        <%= value %>
      </li>
      <% })%>
  </ul>

</body>

</html>

```
